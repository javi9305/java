public static void main(String[] args) {
        //Declaracion de variables asignandoles un valor.
       int a = 3, b = 2;
       int c = a + 5 - b; //Se realiaza la operacion aritmetica y el resultado es asignado a una nueva variable.
       
        //Impresion del resultado de la variable c.
        System.out.println("c = " + c);
       
        //Asignacion de valores junto con una variable.
        // += -> Operador de composicion.
       a+= 1; //a = a + 1;
        System.out.println("a = " + a);
       
        a += 3; //a = a + 3;
        System.out.println("a = " + a);
       
        a -= 2; //a = a - 2
        System.out.println("a = " + a);
        
        a *= 4; //a = a = a * 4
        System.out.println("a = " + a);
        
        a /= 5; // a = a / 5
        System.out.println("a = " + a);
        
        a %= 2; // a = a % 2
        System.out.println("a = " + a);
    }
